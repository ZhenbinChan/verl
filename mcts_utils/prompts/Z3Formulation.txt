from z3 import *

# Declare sorts
Person = DeclareSort('Person')
Region = DeclareSort('Region')

# Declare predicates
Black = Function('Black', Person, BoolSort())
Westernized = Function('Westernized', Person, BoolSort())
FromRegion = Function('FromRegion', Person, Region, BoolSort())
SaltSufficientHistory = Function('SaltSufficientHistory', Region, BoolSort())
SaltDeficientHistory = Function('SaltDeficientHistory', Region, BoolSort())
GeneticAdaptationToLowSalt = Function('GeneticAdaptationToLowSalt', Person, BoolSort())
HighSaltDiet = Function('HighSaltDiet', Person, BoolSort())
Hypertension = Function('Hypertension', Person, BoolSort())

# Axioms from context and hypothesis
x = Const('x', Person)
r = Const('r', Region)
axioms = [
    # Axiom 1: People from regions with sufficient salt history lack genetic adaptation
    ForAll([x, r], Implies(And(FromRegion(x, r), SaltSufficientHistory(r)), 
                           Not(GeneticAdaptationToLowSalt(x)))),
    # Axiom 2: Black people from regions with salt deficiency have genetic adaptation
    ForAll([x, r], Implies(And(Black(x), FromRegion(x, r), SaltDeficientHistory(r)), 
                           GeneticAdaptationToLowSalt(x))),
    # Axiom 3: Westernized individuals have high-salt diet
    ForAll([x], Implies(Westernized(x), HighSaltDiet(x))),
    # Hypothesis H: For westernized black individuals, hypertension iff high-salt diet AND genetic adaptation
    ForAll([x], Implies(And(Black(x), Westernized(x)), 
                        Hypertension(x) == And(HighSaltDiet(x), GeneticAdaptationToLowSalt(x))))
]
